import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
from sklearn.model_selection import train_test_split, cross_val_score
from sklearn.tree import DecisionTreeClassifier, plot_tree
from sklearn.preprocessing import LabelEncoder, StandardScaler

df = pd.read_csv('bitacora_de_cambios_tr_indice_cla_mensual_nac_2007_2023.csv', encoding="utf-8")

# Convertir "Fecha de modificacin" a tipo fecha
df["Fecha de modificacin"] = pd.to_datetime(df["Fecha de modificacin"], errors='coerce', dayfirst=True)

df = df.dropna(subset=["Nmero de registro o fila", "Unnamed: 6"])

X = df.drop(columns=["Estatus del cambio", "Nombre de campo"])  
y = df["Nombre de campo"]

# Codificacin de la variable objetivo
encoder = LabelEncoder()
y_encoded = encoder.fit_transform(y)

# Normalizar las variables numricas
scaler = StandardScaler()
X_scaled = pd.DataFrame(scaler.fit_transform(X.select_dtypes(include=[np.number])), columns=X.select_dtypes(include=[np.number]).columns)

# Dividir datos en entrenamiento y prueba
X_train, X_test, y_train, y_test = train_test_split(X_scaled, y_encoded, test_size=0.2, stratify=y_encoded, random_state=42)

# Obtener valores ptimos de `ccp_alpha` usando validacin cruzada
clf = DecisionTreeClassifier(random_state=42)
path = clf.cost_complexity_pruning_path(X_train, y_train)
ccp_alphas = path.ccp_alphas[:-1]  # Omitimos el ltimo ya que poda todo el rbol

# Evaluar cada valor de `ccp_alpha` usando validacin cruzada
scores = []
for ccp_alpha in ccp_alphas:
    tree = DecisionTreeClassifier(random_state=42, ccp_alpha=ccp_alpha)
    score = np.mean(cross_val_score(tree, X_train, y_train, cv=5))
    scores.append(score)

# Seleccionar el mejor `ccp_alpha`
best_alpha = ccp_alphas[np.argmax(scores)]

# Entrenar el rbol con el mejor `ccp_alpha`
tree = DecisionTreeClassifier(random_state=42, ccp_alpha=best_alpha)
tree.fit(X_train, y_train)

# Visualizar el rbol de decisin
